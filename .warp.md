Quick commands for AFL Fantasy Intelligence Platform development.

## Build & Run Commands

```bash
# 🏗️ Clean Build
clean_build:
  name: "🏗️ AFL Fantasy Clean Build"  
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && ./scripts/build.sh"
  description: "Clean build + unit tests for AFL Fantasy iOS"

# 📱 Run Simulator  
run_simulator:
  name: "📱 Run AFL Fantasy Simulator"
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && ./scripts/run-simulator.sh"
  description: "Launch AFL Fantasy in iPhone 15 simulator"

# 🧪 Run Tests
run_tests:
  name: "🧪 AFL Fantasy Tests"
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && ./scripts/run-tests.sh"  
  description: "Run unit & UI tests"

# 📂 Open Xcode Project
open_xcode:
  name: "📂 Open AFL Fantasy Xcode"
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && open ios/AFLFantasy.xcodeproj"
  description: "Open project in Xcode"

# 🔍 Project Status
project_status:
  name: "🔍 AFL Fantasy Status"
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && echo '📁 Project: AFL Fantasy Intelligence Platform iOS' && pwd && git status --porcelain && echo '📊 Build Info:' && xcodebuild -project ios/AFLFantasy.xcodeproj -list 2>/dev/null || echo 'Project not yet created'"
  description: "Show project status and git state"
```

## Development Shortcuts

```bash
# ⚡ Quick Build & Run
quick_run:
  name: "⚡ AFL Fantasy Quick Run"
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && ./scripts/build.sh && open -a Simulator"
  description: "Build and launch simulator in one command"

# 🗂️ Show Project Structure  
show_structure:
  name: "🗂️ Show AFL Fantasy Structure"
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && tree -L 3 -I 'build|DerivedData|.git' || find . -type d -depth 3 2>/dev/null | head -20"
  description: "Display project directory structure"

# 📝 Show Recent Commits
show_commits:
  name: "📝 AFL Fantasy Recent Commits"  
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && git log --oneline -10 --graph"
  description: "Show last 10 commits with graph"
```

## iOS Specific

```bash
# 📲 List Simulators
list_simulators:
  name: "📲 List iOS Simulators"
  command: "xcrun simctl list devices available --json | jq -r '.devices | to_entries[] | select(.key | contains(\"iOS\")) | .key as $version | .value[] | select(.isAvailable==true) | \"\\($version): \\(.name) (\\(.udid))\"' 2>/dev/null || xcrun simctl list devices available"
  description: "Show available iOS simulators"

# 🧹 Clean Simulators
clean_simulators:
  name: "🧹 Clean iOS Simulators"  
  command: "xcrun simctl shutdown all && xcrun simctl erase all"
  description: "Reset all iOS simulators"
  warning: "This will delete all simulator data"

# 📦 Archive Build
archive_build:
  name: "📦 AFL Fantasy Archive"
  command: "cd ~/workspace/10_projects/afl_fantasy_ios && xcodebuild -project ios/AFLFantasy.xcodeproj -scheme AFLFantasy -archivePath build/AFLFantasy.xcarchive archive"
  description: "Create distribution archive"
```
